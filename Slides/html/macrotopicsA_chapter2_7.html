def resid_three_period2(a3,a2,e2,params):

    # 
    # (cash on hand)
    coh = (1+params.rent)*a2 + params.y2*e2
    
    if coh - a3 > 0.0: # 
        mu2 = mu_CRRA(coh-a3,params.)
    else:
        mu2 = 10000.0
    
    # 
    mu3 = mu_CRRA((1 + params.rent)*a3 + params.y3, params.)

    # (residual)
    return params.*(1.0 + params.rent)*(mu3/mu2) - 1.0

def resid_three_period1(a2,a1,e1,a2_nl,params):

    # 
    # (cash on hand)
    coh = (1 + params.rent)*a1 + params.y1*params.endow[e1]

    if coh - a2 > 0.0: # 
        mu1 = mu_CRRA(coh-a2,params.)
    else:
        mu1 = 10000.0

    mu2 = np.zeros(params.ny)
    for i in range(params.ny):

        # 
        interp1 = interpolate.interp1d(params.grid_a,a2_nl[:,i])
        cons = interp1(a1)
        
        # 
        mu2[i] = mu_CRRA(cons, params.)

    # (residual)
    exp_val = np.dot(params.tran[e1, :], mu2)

    return params.*(1.0 + params.rent)*(exp_val/mu1) - 1.0
